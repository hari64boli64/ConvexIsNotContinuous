\newcommand{\DrawZigzagArrow}[4][]{%
  \draw[->, rounded corners, #1] (#2) -- ($ (#2)!0.5!(#2|-#3) $) -- ($ (#3)!0.5!(#3|-#2) $) -- (#3);
  \ifnum#4=1
    \node[circle,fill=green!30!white] at ($ (#2)!0.5!(#3) $) {Yes}
  \else
    \node[circle,fill=red!30!white] at ($ (#2)!0.5!(#3) $) {No}
  \fi
}
